45 bool add_range(uint32_t lo, uint32_t hi) {
46 if (hi < lo) return false;
47
48 {
49 auto it = ranges.find(UnicodeRange(lo, hi));
50 if (it != ranges.end() && it->lo <= lo && hi <= it->hi) return false;
51 }
52
53 // Look for an inmediate range to the left of [lo, hi]. If exists
54 // then erase and extend [lo, hi] accordingly
55 if (lo > 0) {
56 auto it = ranges.find(UnicodeRange(lo - 1, lo - 1));
57 if (it != ranges.end()) {
58 lo = it->lo;
59 if (it->hi > hi) hi = it->hi;
60 ranges.erase(it);
61 }
62 }
63
64 // Look for an inmediate range to the right of [lo, hi]. If exists
65 // then erase and extend [lo, hi] accordingly
66 // Note: The last unicode code point is 0x10FFFF
67 if (hi < UTF8MAX) {
68 auto it = ranges.find(UnicodeRange(hi + 1, hi + 1));
69 if (it != ranges.end()) {
70 hi = it->hi;
71 // Not necesary to check for lo. Did it in previous step.
72 ranges.erase(it);
73 }
74 }
75
76 // Search for ranges inside [lo, hi] and erase them
77 // The remaining ranges that overlap with [lo, hi] must be inside.
78 for (;;) {
79 auto it = ranges.find(UnicodeRange(lo, hi));
80 if (it == ranges.end()) break;
81 ranges.erase(it);
82 }
83
84 // Add [lo, hi]
85 ranges.insert(UnicodeRange(lo, hi));
86 return true;
87 }

91 void negate() {
92 std::vector<UnicodeRange> new_ranges;
93 new_ranges.reserve(ranges.size() + 1);
94
95 auto it = ranges.begin();
96 if (it == ranges.end()) {
97 new_ranges.push_back(UnicodeRange(0, UTF8MAX));
98 } else {
99 uint32_t next_lo = 0;
100 if (it->lo == 0) {
101 next_lo = it->hi + 1;
102 ++it;
103 }
104 for (; it != ranges.end(); ++it) {
105 new_ranges.push_back(UnicodeRange(next_lo, it->lo - 1));
106 next_lo = it->hi + 1;
107 }
108 if (next_lo <= UTF8MAX)
109 new_ranges.push_back(UnicodeRange(next_lo, UTF8MAX));
110 }
111
112 ranges.clear();
113 for (auto &range : new_ranges) {
114 ranges.insert(range);
115 }

118 // Build the automaton for a range of two bytes and stores it in the lva_ptr
119 void buildTwoBytesAutomaton(char lo[2], char hi[2]) {
120 if (lo[0] == hi[0]) {
121 // Same first byte
122 auto A1 = std::make_unique<LogicalVA>(CharClass(lo[0]));
123 auto A2 = std::make_unique<LogicalVA>(CharClass({lo[1], hi[1]}));
124 A1->cat(*A2);
125 lva_ptr = std::move(A1);
126 } else {
127 // Different first byte
128 auto A1 = std::make_unique<LogicalVA>(CharClass(lo[0]));
129 auto A2 = std::make_unique<LogicalVA>(CharClass({lo[1], '\xBF'}));
130 A1->cat(*A2);
131 auto B1 = std::make_unique<LogicalVA>(CharClass(hi[0]));
132 auto B2 = std::make_unique<LogicalVA>(CharClass({'\x80', hi[1]}));
133 B1->cat(*B2);
134 A1->alter(*B1);
135 if (lo[0] + 1 < hi[0]) {
136 // There are more bytes between the first bytes
137 auto C1 = std::make_unique<LogicalVA>(CharClass({char(lo[0] + 1), char(hi[0] - 1)}));
138 auto C2 = std::make_unique<LogicalVA>(CharClass({'\x80', '\xBF'}));
139 C1->cat(*C2);
140 A1->alter(*C1);
141 }
142 lva_ptr = std::move(A1);
143 }

147 void buildThreeBytesAutomaton(char lo[3], char hi[3]) {

173 auto B1 = std::make_unique<LogicalVA>(CharClass(hi[0]));
174 auto B2 = std::make_unique<LogicalVA>(CharClass(hi[1]));

194 hi_mid[1] = '\xBF';
195 } else {
196 hi_mid[0] = hi[0];
197 hi_mid[1] = char(hi[1] - 1);

308 // Build the automaton for a character class and stores it in the lva_ptr
309 void buildCharacterClassAutomaton() {
310 auto it = ranges.begin();
311 std::stack<UnicodeRange> stack;
312 stack.push(*it);
313 // 1 byte automaton
314 {
315 CharClass ccb;
316 while (!stack.empty()) {
317 UnicodeRange r = stack.top();
318 if (r.lo > 0x7F) break;
319 stack.pop();
320 if (r.hi <= 0x7F) {
321 // Handle range
322 ccb.add_range(r.lo, r.hi);
323 if (stack.empty() && ++it != ranges.end()) stack.push(*it);
324 } else {
325 // Split range
326 stack.emplace(0x80, r.hi);
327 stack.emplace(r.lo, 0x7F);
328 }
329 }
330 if (!ccb.empty()) {
331 lva_ptr = std::make_unique<LogicalVA>(ccb);
332 }
333 }
334 // 2 bytes automaton
335 while (!stack.empty()) {
336 UnicodeRange r = stack.top();
337 if (r.lo > 0x7FF) break;
338 stack.pop();
339 if (r.hi <= 0x7FF) {
340 // Handle range
341 char lo[2] = {char((r.lo >> 6) | 0xC0), char((r.lo & 0x3F) | 0x80)};
342 char hi[2] = {char((r.hi >> 6) | 0xC0), char((r.hi & 0x3F) | 0x80)};
343 if (lva_ptr != nullptr) {
344 auto A = std::move(lva_ptr);
345 buildTwoBytesAutomaton(lo, hi);
346 lva_ptr->alter(*A);
347 } else {
348 buildTwoBytesAutomaton(lo, hi);
349 }
350 if (stack.empty() && ++it != ranges.end()) stack.push(*it);
351 } else {
352 // Split range
353 stack.emplace(0x800, r.hi);
354 stack.emplace(r.lo, 0x7FF);
355 }
356 }
357 // 3 bytes automaton
358 while (!stack.empty()) {
359 UnicodeRange r = stack.top();
360 if (r.lo > 0xFFFF) break;
361 stack.pop();
362 if (r.hi <= 0xFFFF) {
363 // Handle range
364 char lo[3] = {char((r.lo >> 12) | 0xE0), char(((r.lo >> 6) & 0x3F) | 0x80), char((r.lo & 0x3F) | 0x80)};
365 char hi[3] = {char((r.hi >> 12) | 0xE0), char(((r.hi >> 6) & 0x3F) | 0x80), char((r.hi & 0x3F) | 0x80)};
366 if (lva_ptr != nullptr) {
367 auto A = std::move(lva_ptr);
368 buildThreeBytesAutomaton(lo, hi);
369 lva_ptr->alter(*A);
370 } else {
371 buildThreeBytesAutomaton(lo, hi);
372 }
373 if (stack.empty() && ++it != ranges.end()) stack.push(*it);
374 } else {
375 // Split range
376 stack.emplace(0x10000, r.hi);
377 stack.emplace(r.lo, 0xFFFF);
378 }
379 }
380 // 4 bytes automaton
381 while (!stack.empty()) {
382 UnicodeRange r = stack.top();
383 stack.pop();
384 // Handle range
385 char lo[4] = {char((r.lo >> 18) | 0xF0), char(((r.lo >> 12) & 0x3F) | 0x80), char(((r.lo >> 6) & 0x3F) | 0x80), char((r.lo & 0x3F) | 0x80)};
386 char hi[4] = {char((r.hi >> 18) | 0xF0), char(((r.hi >> 12) & 0x3F) | 0x80), char(((r.hi >> 6) & 0x3F) | 0x80), char((r.hi & 0x3F) | 0x80)};
387 if (lva_ptr != nullptr) {
388 auto A = std::move(lva_ptr);
389 buildFourBytesAutomaton(lo, hi);
390 lva_ptr->alter(*A);
391 } else {
392 buildFourBytesAutomaton(lo, hi);
393 }
394 if (stack.empty() && ++it != ranges.end()) stack.push(*it);
395 }
396 }

537 A1->alter(*C1);
538 A1->alter(*D1);

541

545 else if (ctx->TAB()) {
546 lva_ptr = std::make_unique<LogicalVA>(CharClass('\t'));
